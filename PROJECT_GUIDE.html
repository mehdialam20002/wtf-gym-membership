Controller Example — Gym Create Karna

// backend/src/controllers/gymController.js

async function createGym(req, res) {
  // 1. Frontend se data lo
  const { name, area, city, address, phone, hours } = req.body;

  // 2. Slug banao (URL-friendly naam)
  const slug = name.toLowerCase().replace(/\s+/g, '-');

  // 3. Database mein save karo
  const gym = await prisma.gym.create({
    data: { name, slug, area, city, address, phone, hours }
  });

  // 4. Frontend ko response bhejo
  res.status(201).json(gym);
}

Transaction Example — Plan-Gym Assignment

// PUT /api/plans/:id/gyms
async function updatePlanGyms(req, res) {
  const { id } = req.params;         // plan ID
  const { gymIds } = req.body;       // [gymId1, gymId2, ...]

  // Transaction: ya dono ho ya kuch nahi
  await prisma.$transaction([
    // Step 1: Purane assignments delete karo
    prisma.gymPlan.deleteMany({ where: { planId: id } }),

    // Step 2: Nayi assignments banao
    prisma.gymPlan.createMany({
      data: gymIds.map(gymId => ({ planId: id, gymId }))
    })
  ]);

  res.json({ success: true });
}
Kyun Transaction? Agar Step 1 success ho aur Step 2 fail ho → gym ke koi plans nahi! Transaction ensure karta hai: ya dono steps complete honge, ya kuch bhi nahi.

Authentication Flow — Login kaise kaam karta hai

JWT tokens se stateless auth

Step-by-Step Auth Flow

1
Admin login karta hai — Email + Password form submit karta hai
2
Backend verify karta hai — bcrypt.compare(input, hashedPassword)
3
JWT token generate hota hai — jwt.sign({ adminId }, secret, { expiresIn: '7d' })
4
Frontend store karta hai — localStorage.setItem('gym_admin_token', token)
5
Agle requests mein — Axios interceptor: Authorization: Bearer <token> header add karta hai
6
Backend verify karta hai — protect middleware: jwt.verify(token, secret)
// Middleware — JWT verify karo
function protect(req, res, next) {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'No token' });

  const decoded = jwt.verify(token, process.env.JWT_SECRET);
  req.admin = decoded;  // admin info agle function ko milegi
  next();             // agle function pe jaao
}

// Use karo is tarah:
router.post('/', protect, createGym);
// pehle protect chalega, phir createGym

JWT Token kya hota hai?

// JWT = 3 parts, Base64 encoded, dot se separate
eyJhbGciOiJIUzI1NiJ9      ← Header (algorithm: HS256)
.eyJhZG1pbklkIjoiMSJ9     ← Payload (data: adminId)
.xyzABC123signature        ← Signature (tamper-proof)

// Payload mein store karo:
{ adminId: 'abc123', iat: 1234567, exp: 1241234 }
// iat = issued at, exp = expires at

// Koi bhi decode kar sakta hai payload ko
// Lekin signature fake nahi kar sakta (secret key chahiye)
// Isliye sensitive data (password) kabhi payload mein mat rakho!

Hooks & State Management

Data kaise fetch hota hai aur kahan store hota hai

Custom Hook — useGyms()

// frontend/src/hooks/useGyms.js

function useGyms(adminMode = false) {
  const [gyms, setGyms] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const url = adminMode ? '/gyms/all' : '/gyms';
    api.get(url).then(r => {
      setGyms(r.data);
      setLoading(false);
    });
  }, [adminMode]);

  async function createGym(data) {
    const res = await api.post('/gyms', data);
    setGyms(prev => [...prev, res.data]); // No refetch — local update
    return res.data;
  }

  return { gyms, loading, createGym, /* updateGym, deleteGym... */ };
}

// Koi bhi component use kar sakta hai:
const { gyms, loading } = useGyms(false);   // public
const { gyms, loading } = useGyms(true);    // admin
Kyun Custom Hook? Code reuse. LandingPage aur DashboardPage dono gyms chahiye. Hook ke bina same code dono jagah likhna padta. Hook ek jagah likho, kahin bhi use karo.

Axios Setup — api.js

// frontend/src/services/api.js

const api = axios.create({ baseURL: '/api' });

// Request interceptor: Har request mein token add karo
api.interceptors.request.use(config => {
  const token = localStorage.getItem('gym_admin_token');
  if (token) config.headers.Authorization = `Bearer ${token}`;
  return config;
});

// Response interceptor: 401 aaye to logout karo
api.interceptors.response.use(
  res => res,  // success: as-is return karo
  err => {
    if (err.response?.status === 401) {
      localStorage.removeItem('gym_admin_token');
      window.location.href = '/admin/login';
    }
    return Promise.reject(err);
  }
);

AuthContext — Global Login State

// App load hote hi check karo: token hai?
useEffect(() => {
  const token = localStorage.getItem('gym_admin_token');
  if (token) {
    api.get('/admin/me')
      .then(r => { setAdmin(r.data); setIsAuthenticated(true); })
      .catch(() => localStorage.removeItem('gym_admin_token'));
  }
}, []);

// Context provide karo poore app ko:
<AuthContext.Provider value={{ admin, isAuthenticated, login, logout }}>
  {children}
</AuthContext.Provider>

15 Interview Questions — With Answers

Basic se Advanced tak — is project ke basis pe

Basic
Q1: Frontend aur Backend alag kyun rakhe? Ek hi codebase mein kyun nahi?

Separation of Concerns — Frontend ka kaam dikhana hai (UI), Backend ka kaam data manage karna hai (logic + database).

  • Alag teams parallel kaam kar sakti hain
  • Same backend → multiple frontends (website + mobile app)
  • Deploy alag alag ho sakti hain (frontend pe CDN, backend pe server)
  • Security: Database credentials sirf backend pe, browser mein kabhi nahi
Basic
Q2: Slug kya hota hai aur kyun use kiya?

Slug = URL-friendly version of a name

"WTF Gym Noida Sector 18" → "wtf-gym-noida-sector-18"

  • /gym/123 → Bad: user ko pata nahi kya hai
  • /gym/wtf-gym-noida-sector-18 → Good: readable + SEO friendly
  • Database mein unique constraint lagaya hai — duplicate slugs nahi ho sakte
Basic
Q3: Amenities aur Benefits JSON mein kyun store kiye? Alag table kyun nahi?

Simple strings hain, koi relation nahi hai inki kisi doosri table se. Alag table banane ki zarurat nahi thi.

  • JSON array use karo jab: Simple list ho, koi relation nahi, fast read chahiye
  • Alag table banao jab: Benefits ke saath kuch aur data ho (image, icon, link), ya benefits filter/search karne ho

Is case mein: ["Gym Access", "Trainer Support"] → JSON kaafi hai

Basic
Q4: isActive aur isVisible mein kya fark hai?

isActive → Gym ka flag (gym publicly dikhe ya nahi)

isVisible → Plan ka flag (plan publicly dikhe ya nahi)

Dono alag models mein same concept apply kiya. Admin kisi gym ya plan ko temporarily hide kar sakta hai bina delete kiye. Delete permanent hai, hide reversible hai.

Basic
Q5: GymPlan junction table kyun chahiye? Direct gym mein plan IDs store karo na?

Many-to-Many relationship hai. Ek gym ke multiple plans, ek plan multiple gyms pe.

  • Wrong: Gym table mein planIds: ["p1","p2"] store karo → Plan price badlo → Har gym mein manually update karna padega
  • Right: GymPlan junction table → Plan ek jagah update karo → Automatically sab gyms pe reflect
  • Unique constraint: Same plan twice ek gym pe assign nahi ho sakta
  • Cascade delete: Gym delete karo → GymPlan rows automatically delete
Intermediate
Q6: Prisma transaction kyun use kiya plan-gym assignment mein?

Atomicity — Ya sab kuch hoga ya kuch nahi (ACID ka 'A')

Without transaction:

  • Step 1: Purane assignments delete ✅
  • Step 2: Nayi assignments create ❌ (error!)
  • Result: Plan ke koi gyms nahi! Broken state.

With transaction:

  • Step 1 fail ya Step 2 fail → dono rollback
  • Database consistent state mein rehta hai
Intermediate
Q7: Axios interceptors kaise kaam karte hain? Kyun use kiya?

Interceptors = middleware for HTTP requests/responses

  • Request interceptor: Har request se pehle run hota hai → token add karta hai automatically
  • Response interceptor: Har response ke baad run hota hai → 401 aaye to logout + redirect

Bina interceptors ke: Har API call mein manually token add karna padta, har 401 manually handle karna padta. 50 API calls → 50 jagah same code.

Interceptors ke saath: Ek jagah likho, poore app mein apply.

Intermediate
Q8: useEffect mein [slug] dependency kyun diya?

useEffect(() => { fetchGym(slug) }, [slug])

  • [] (empty): Sirf ek baar run hoga (component mount pe) → Wrong! /gym/abc se /gym/xyz navigate karo → naya data nahi aayega
  • [slug]: Jab bhi slug change ho → effect dobara run hoga → naya gym data fetch hoga
  • no array: Har render pe run hoga → Infinite loop!
Intermediate
Q9: JWT token localStorage mein store karna safe hai?

Partially safe. Tradeoff hai:

  • localStorage risk: XSS attack mein JavaScript token read kar sakta hai
  • Better option: HttpOnly Cookie → JavaScript access nahi kar sakta → XSS se safe
  • Is project mein localStorage use kiya → Simpler implementation, admin-only app mein acceptable
  • Production grade app mein: HttpOnly cookies prefer karein
Intermediate
Q10: Rate limiting kyun important hai? Sirf login pe kyun alag limit?

Brute force attacks se bachne ke liye:

  • Without rate limiting: Script → 1000 passwords/second → account hack possible
  • With 10/15min: Practically impossible
  • Login pe strict (10/15min) → Sensitive! Wrong password → account compromise risk
  • General API pe loose (100/15min) → Normal usage support karo, abuse rokko
Advanced
Q11: Is project mein N+1 query problem ho sakti hai? Kahan aur kaise solve karein?

N+1 problem: 1 query se gyms lo, phir har gym ke liye alag query plans ke liye → 1 + N queries

Is project mein: Prisma include use kiya → JOIN karta hai → sirf 1 query

prisma.gym.findMany({
  include: { gymPlans: { include: { plan: true } } }
})
// Ek hi query mein sab kuch — N+1 problem nahi!
Advanced
Q12: Soft delete vs Hard delete — kya use karna chahiye tha?

Hard delete = DELETE FROM table (is project mein use kiya)

Soft delete = isDeleted = true, data rehta hai

  • Soft delete ke fayde: History preserve, restore possible, audit trail
  • Is project mein hard delete theek hai kyunki: Gym business simple, history ki zarurat nahi, complexity badh jaati
  • Soft delete kab use karein: Financial data, user accounts, compliance-heavy systems
Advanced
Q13: Agar 1 lakh users ek saath aaye to kya hoga? Kaise handle karein?

Current architecture: Single server → bottleneck at high load

  • Load Balancer → Multiple backend instances
  • Database Read Replicas → Reads alag server pe, writes master pe
  • Redis Cache → Gym list frequently accessed → cache 5 min ke liye
  • CDN → Static assets (JS, CSS, images) edge pe serve karo
  • Stateless JWT → Is project ka JWT approach horizontally scale kar sakta hai (no session state)
Advanced
Q14: Custom hooks mein error handling kaise improve karein?

Current: Basic error state. Better approach:

  • React Query / TanStack Query use karo → Automatic caching, background refetch, retry logic, loading/error states built-in
  • Error boundaries → Component crash hone pe fallback UI dikhao
  • Toast notifications → User ko meaningful error messages
  • Retry logic → Network error pe automatically retry
Advanced
Q15: Is project ko production mein deploy kaise karoge?
  • Frontend: npm run build → Vercel / Netlify (free, CDN included)
  • Backend: Railway / Render / DigitalOcean → Node.js app
  • Database: Supabase / Railway PostgreSQL → prisma migrate deploy
  • Secrets: .env file mein → DATABASE_URL, JWT_SECRET (kabhi git mein push mat karo)
  • CI/CD: GitHub Actions → Push karo → Automatic deploy
  • Admin user: Seed script se manually create karo (registration page nahi hai intentionally)
